const PI = 3.14;

var d = parseFloat(prompt("Diameter of circle?", ""));

//The area of a circle is technically pi times radius^2, but we are currently dealing with diameter. Just divide diameter by 2 first.

function areaCircle(d) {
	return (PI) * ((d / 2) * (d / 2));
}

alert("The area of a circle with diameter " + d + " is " + areaCircle(d) + " units.");
