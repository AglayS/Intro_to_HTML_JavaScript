/* MY PLAN: First, declare my variables. Two will do, i and j. Next, initialize with for (using while breaks it), and write out conditions for variable i (don't forget increments!). 
Next, make a nested for loop with variable j and write out conditions for that. Under the inner for loop should be something that displays '*', and a line break somewhere afterwards. */ 

var i; //First variable declared

	for (var j = 1; j <= i; j++) { //7 rows of asterisks with increments

/* MY PLAN: Variable i is already declared, and it won't affect the first task, so no worries there. Because it's 1-100,

AND ALSO, if something can be divided by both 3 and 5 to put FizzBuzz. */


for (var i = 1; (i < 101) && (i >= 1); i++) { //For var i, must be 1-100.
	
	if (i % 15 == 0) { //For when i can be divided by both 3 and 5.

	}
	else if (i % 3 == 0) { //For dividing by 3.
	
	document.write("Fizz<br />");
